@login_required
def payment_page(request):
    student = Student.objects.get(user=request.user)
    fee = Fee.objects.get(student=student)

    balance_due = fee.total_amount - fee.paid_amount

    return render(request, 'payments/payment_page.html', {
        'fee': fee,
        'balance_due': balance_due
    })

@login_required
def payment_cancel(request):
    order_id = request.GET.get("payment_id")

    if order_id:
        payment = Payment.objects.get(transaction_id=order_id)
        payment.status = "FAILED"
        payment.save()

    return render(request, 'payments/cancel.html')


import razorpay
from django.conf import settings

client = razorpay.Client(auth=(settings.RAZORPAY_KEY_ID, settings.RAZORPAY_KEY_SECRET))

@login_required
def create_order(request):
    student = Student.objects.get(user=request.user)
    fee = Fee.objects.get(student=student)
    fee_type = request.POST.get("fee_type")
    if not fee_type:
        return JsonResponse({"error": "Fee type required"}, status=400)
    amount = int(request.POST.get("amount")) * 100  # paise

    order = client.order.create({
        "amount": amount,
        "currency": "INR",
        "payment_capture": "1"
    })

    # Save payment (PENDING)
    payment = Payment.objects.create(
        user=request.user,
        fee=fee,
        amount=amount / 100,
        payment_type=fee_type,
        transaction_id=order['id'],
        status="PENDING",
        payment_method="RAZORPAY"
    )
    return JsonResponse({
        "order_id": order['id'],
        "amount": amount,
        "key": settings.RAZORPAY_KEY_ID
    })

@login_required
def razorpay_success(request):
    order_id = request.GET.get("payment_id")

    payment = Payment.objects.get(transaction_id=order_id)

    payment.status = "SUCCESS"
    payment.paid_at = timezone.now()
    payment.save()

    fee = payment.fee
    fee.paid_amount += float(payment.amount)
    if fee.paid_amount < fee.total_amount:
        fee.status = "PARTIAL"
    else:
        fee.status = "PAID"

    fee.save()

    return render(request, 'payments/success.html', {'payment': payment})

from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
from reportlab.lib.styles import getSampleStyleSheet
from django.http import HttpResponse
from django.shortcuts import get_object_or_404

@login_required
def download_receipt(request, payment_id):
    payment = get_object_or_404(Payment, id=payment_id)

    response = HttpResponse(content_type='application/pdf')
    response['Content-Disposition'] = f'attachment; filename="receipt_{payment.transaction_id}.pdf"'

    doc = SimpleDocTemplate(response)
    styles = getSampleStyleSheet()

    elements = []

    elements.append(Paragraph("Payment Receipt", styles['Title']))
    elements.append(Spacer(1, 12))

    elements.append(Paragraph(f"Student: {payment.user.username}", styles['Normal']))
    elements.append(Paragraph(f"Amount Paid: ₹{payment.amount}", styles['Normal']))
    elements.append(Paragraph(f"Transaction ID: {payment.transaction_id}", styles['Normal']))
    elements.append(Paragraph(f"Payment Method: {payment.payment_method}", styles['Normal']))
    elements.append(Paragraph(f"Date: {payment.paid_at}", styles['Normal']))

    doc.build(elements)

    window.location.href = "{% url 'student_dashboard' %}";

document.getElementById("pay-btn").onclick = function () {

    let amount = document.getElementById("amount").value;
    let feeType = document.getElementById("fee_type").value; // ✅ get inside function
    let maxAmount = {{ balance_due }};
    let errorBox = document.getElementById("error-msg");

    errorBox.innerHTML = "";

    if (!amount || amount <= 0) {
        errorBox.innerHTML = "Please enter a valid amount.";
        return;
    }

    if (amount > maxAmount) {
        errorBox.innerHTML = "Amount cannot exceed balance.";
        return;
    }

    fetch("{% url 'create_order' %}", {
        method: "POST",
        headers: {
            "Content-Type": "application/x-www-form-urlencoded",
            "X-CSRFToken": "{{ csrf_token }}"
        },
        body: new URLSearchParams({
            amount: amount,
            fee_type: feeType   // ✅ FIXED (VERY IMPORTANT)
        })
    })
    .then(res => res.json())
    .then(data => {

        var options = {
            key: data.key,
            amount: data.amount,
            currency: "INR",
            name: "EduTrack Fee Payment",
            description: feeType + " Payment", // optional improvement
            order_id: data.order_id,

            handler: function (response) {
                window.location.href = "/payment-success/?payment_id=" + data.order_id;
            },

            modal: {
                ondismiss: function () {
                    window.location.href = "/payment-cancel/?payment_id=" + data.order_id;
                }
            },

            theme: {
                color: "#0d7377"
            }
        };

        var rzp = new Razorpay(options);
        rzp.open();
    });
};
